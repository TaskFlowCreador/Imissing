        let activeLineIndex = -1;

        // SISTEMA DE ACTUALIZACIÓN Y LLENADO DINÁMICO (PINTADO)
        function updateKaraokeAndPainting(currentTime) {
            let currentLineIndex = -1;
            
            for (let i = 0; i < lyricsData.length; i++) {
                if (currentTime >= lyricsData[i].start && (i === lyricsData.length - 1 || currentTime < lyricsData[i + 1].start)) {
                    currentLineIndex = i;
                    break;
                }
            }

            if (currentLineIndex !== -1) {
                // Si cambió de línea activa, reajustamos clases
                if (currentLineIndex !== activeLineIndex) {
                    if (activeLineIndex !== -1) {
                        const prevEl = document.getElementById(`lyric-${activeLineIndex}`);
                        if (prevEl) prevEl.classList.remove('active');
                    }
                    activeLineIndex = currentLineIndex;
                    const activeEl = document.getElementById(`lyric-${activeLineIndex}`);
                    if (activeEl) {
                        activeEl.classList.add('active');
                        activeEl.scrollIntoView({ behavior: 'smooth', block: 'center' });
                    }
                }

                // Cálculo del porcentaje exacto de avance de la línea actual para ir pintando las letras
                const activeLine = lyricsData[currentLineIndex];
                const durationLine = activeLine.end - activeLine.start;
                const progressLocal = currentTime - activeLine.start;
                let pct = (progressLocal / durationLine) * 100;
                if (pct > 100) pct = 100;
                if (pct < 0) pct = 0;

                const activeEl = document.getElementById(`lyric-${currentLineIndex}`);
                if (activeEl) {
                    activeEl.style.setProperty('--paint-progress', `${pct}%`);
                }
            }
        }

                this.speedX = isExplosion ? (Math.random() - 0.5) * 5 : (Math.random() - 0.5) * 1;
                this.speedY = isExplosion ? (Math.random() - 0.5) * 5 : -Math.random() * 1.2 - 0.4;
                this.color = `hsla(${340 + Math.random() * 20}, 90%, 65%, ${Math.random() * 0.3 + 0.3})`;
                this.isExplosion = isExplosion; this.life = 1;
            }
            draw() {
                ctx.save(); ctx.translate(this.x, this.y); ctx.beginPath(); ctx.scale(this.size / 10, this.size / 10);
                ctx.moveTo(0, 0);
                ctx.bezierCurveTo(-5, -5, -10, -2, -10, 3); ctx.bezierCurveTo(-10, 8, -5, 12, 0, 17);
                ctx.bezierCurveTo(5, 12, 10, 8, 10, 3); ctx.bezierCurveTo(10, -2, 5, -5, 0, 0);
                ctx.fillStyle = this.color; ctx.globalAlpha = this.isExplosion ? this.life : 0.6; ctx.fill(); ctx.restore();
            }
            update() {
                this.x += this.speedX; this.y += this.speedY;
                if (this.isExplosion) this.life -= 0.02;
            }
        }

        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            if (particles.length < 40 && Math.random() < 0.03) particles.push(new Particle(Math.random() * canvas.width, canvas.height + 20));
            for (let i = 0; i < particles.length; i++) {
                particles[i].update(); particles[i].draw();
                if (particles[i].y < -30 || (particles[i].isExplosion && particles[i].life <= 0)) { particles.splice(i, 1); i--; }
            }
            requestAnimationFrame(animate);
        }
        animate();
    </script>
